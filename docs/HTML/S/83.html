  30   bool retVal = false;
  31   if (fs == NULL)
  32     return retVal;
  33   // Gives you the info for number of childs with a parent
  34   if (sourceModel()->rowCount(index) > 0) {
  35     for (int nChild = 0; nChild < sourceModel()->rowCount(index); ++nChild) {
  36       QModelIndex childIndex = sourceModel()->index(nChild, 0, index);
  37       if (!childIndex.isValid())
  38         break;
  39       retVal = ShowThis(childIndex);
  40       if (retVal)
  41         break;
  42     }
  43   } else {
  44     QModelIndex useIndex = sourceModel()->index(index.row(), 0, index.parent());
  45     QString path = fs->filePath(useIndex);
  46     path = QDir(store).relativeFilePath(path);

  48     retVal = path.contains(filterRegExp());
