  39 void SingleApplication::receiveMessage() {
  40   QLocalSocket *localSocket = localServer->nextPendingConnection();
  41   if (!localSocket->waitForReadyRead(timeout)) {
  42     qDebug() << localSocket->errorString().toLatin1();
  43     return;
  44   }
  45   QByteArray byteArray = localSocket->readAll();
  46   QString message = QString::fromUtf8(byteArray.constData());
  47   emit messageAvailable(message);
  48   localSocket->disconnectFromServer();
