 413 /**
 414  * @brief ConfigDialog::setProfiles set the profiles and chosen profile from
 415  * MainWindow.
 416  * @param profiles
 417  * @param profile
 418  */
 419 void ConfigDialog::setProfiles(QHash<QString, QString> profiles,
 420                                QString profile) {
 421   // qDebug() << profiles;
 422   if (profiles.contains("")) {
 423     profiles.remove("");
 424     // remove weird "" key value pairs
 425   }
 426 
 427   ui->profileTable->setRowCount(profiles.count());
 428   QHashIterator<QString, QString> i(profiles);
 429   int n = 0;
 430   while (i.hasNext()) {
 431     i.next();
 432     if (!i.value().isEmpty() && !i.key().isEmpty()) {
 433       ui->profileTable->setItem(n, 0, new QTableWidgetItem(i.key()));
 434       ui->profileTable->setItem(n, 1, new QTableWidgetItem(i.value()));
 435       // qDebug() << "naam:" + i.key();
 436       if (i.key() == profile)
 437         ui->profileTable->selectRow(n);
 438     }
 439     ++n;
 440   }
 441 }

 475 /**
 476  * @brief ConfigDialog::on_deleteButton_clicked remove a profile row.
 477  */
 478 void ConfigDialog::on_deleteButton_clicked() {
 479   QSet<int> selectedRows; // we use a set to prevent doubles
 480   QList<QTableWidgetItem *> itemList = ui->profileTable->selectedItems();
 481   if (itemList.count() == 0) {
 482     QMessageBox::warning(this, tr("No profile selected"),
 483                          tr("No profile selected to delete"));
 484     return;
 485   }
 486   QTableWidgetItem *item;
 487   foreach (item, itemList)
 488     selectedRows.insert(item->row());
 489   // get a list, and sort it big to small
 490   QList<int> rows = selectedRows.toList();
 491   qSort(rows.begin(), rows.end());
 492   // now actually do the removing:
 493   foreach (int row, rows)
 494     ui->profileTable->removeRow(row);
 495   if (ui->profileTable->rowCount() < 1)
 496     ui->deleteButton->setEnabled(false);
 497 }

 509 /**

 511  * @todo make this thing more reliable.
 512  */
 513 void ConfigDialog::wizard() {
 514   // mainWindow->checkConfig();
 515   bool clean = false;
 516 
 517   QString gpg = ui->gpgPath->text();
 518   // QString gpg = mainWindow->getGpgExecutable();
 519   if (!QFile(gpg).exists()) {
 520     criticalMessage(
 521         tr("GnuPG not found"),
 522         tr("Please install GnuPG on your system.<br>Install "
 523            "<strong>gpg</strong> using your favorite package manager<br>or <a "
 524            "href=\"https://www.gnupg.org/download/#sec-1-2\">download</a> it "
 525            "from GnuPG.org"));
 526     clean = true;
 527   }
 528 
 529   QStringList names = mainWindow->getSecretKeys();
 530   qDebug() << names;
 531   if (QFile(gpg).exists() && names.empty()) {
 532     KeygenDialog d(this);
 533     if (!d.exec())
 534       return;
 535   }
 536 
 537   QString passStore = ui->storePath->text();
 538 
 539   if (!QFile(passStore).exists()) {
 540     // TODO(annejan) pass version?
 541     if (QMessageBox::question(
 542             this, tr("Create password-store?"),
 543             tr("Would you like to create a password-store at %1?")
 544                 .arg(passStore),
 545             QMessageBox::Yes | QMessageBox::No) == QMessageBox::Yes) {
 546       QDir().mkdir(passStore);
 547 #ifdef Q_OS_WIN
 548       SetFileAttributes(passStore.toStdWString().c_str(),
 549                         FILE_ATTRIBUTE_HIDDEN);
 550 #endif
 551       if (useGit())
 552         mainWindow->executePassGitInit();
 553       mainWindow->userDialog(passStore);
 554     }
 555   }
 556 
 557   if (!QFile(passStore + ".gpg-id").exists()) {
 558     qDebug() << ".gpg-id file does not exist";
 559 
 560     if (!clean) {
 561       criticalMessage(tr("Password store not initialised"),
 562                       tr("The folder %1 doesn't seem to be a password store or "
 563                          "is not yet initialised.")
 564                           .arg(passStore));
 565     }
 566     while (!QFile(passStore).exists()) {
 567       on_toolButtonStore_clicked();
 568       // allow user to cancel
 569       if (passStore == ui->storePath->text())
 570         return;
 571       passStore = ui->storePath->text();
 572     }
 573     if (!QFile(passStore + ".gpg-id").exists()) {
 574       qDebug() << ".gpg-id file still does not exist :/";
 575       // appears not to be store
 576       // init yes / no ?
 577       mainWindow->userDialog(passStore);
 578     }
 579   }
 580 }
