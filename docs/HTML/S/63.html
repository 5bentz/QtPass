  48 /**
  49  * @brief PasswordDialog::setPassword populate the (templated) fields.
  50  * @param password
  51  */
  52 void PasswordDialog::setPassword(QString password) {
  53   QStringList tokens = password.split("\n");
  54   ui->lineEditPassword->setText(tokens[0]);
  55   tokens.pop_front();
  56   if (templating) {
  57     QWidget *previous = ui->checkBoxShow;
  58     for (int i = 0; i < ui->formLayout->rowCount(); ++i) {
  59       QLayoutItem *item = ui->formLayout->itemAt(i, QFormLayout::FieldRole);
  60       if (item == NULL)
  61         continue;
  62       QWidget *widget = item->widget();
  63       for (int j = 0; j < tokens.length(); ++j) {
  64         QString token = tokens.at(j);
  65         if (token.startsWith(widget->objectName() + ':')) {
  66           tokens.removeAt(j);
  67           QString value = token.remove(0, widget->objectName().length() + 1);
  68           reinterpret_cast<QLineEdit *>(widget)->setText(value.trimmed());
  69         }
  70       }
  71       previous = widget;
  72     }
  73     if (allFields) {
  74       for (int j = 0; j < tokens.length(); ++j) {
  75         QString token = tokens.at(j);
  76         if (token.contains(':')) {
  77           int colon = token.indexOf(':');
  78           QString field = token.left(colon);
  79           QString value = token.right(token.length() - colon - 1);
  80           if (!passTemplate.contains(field) && value.startsWith("//"))
  81             continue; // colon is probably from a url
  82           QLineEdit *line = new QLineEdit();
  83           line->setObjectName(field.trimmed());
  84           line->setText(value.trimmed());
  85           ui->formLayout->addRow(new QLabel(field), line);
  86           setTabOrder(previous, line);
  87           previous = line;
  88           tokens.removeAt(j);
  89           --j; // tokens.length() also got shortened by the remove..
  90         }
  91       }
  92     }
  93   }
  94   ui->plainTextEdit->insertPlainText(tokens.join("\n"));
  95 }

  97 /**
  98  * @brief PasswordDialog::getPassword  join the (templated) fields to a QString
  99  * for writing back.
 100  * @return collappsed password.
 101  */
 102 QString PasswordDialog::getPassword() {
 103   QString passFile = ui->lineEditPassword->text() + "\n";
 104   for (int i = 0; i < ui->formLayout->rowCount(); ++i) {
 105     QLayoutItem *item = ui->formLayout->itemAt(i, QFormLayout::FieldRole);
 106     if (item == NULL)
 107       continue;
 108     QWidget *widget = item->widget();
 109     QString text = reinterpret_cast<QLineEdit *>(widget)->text();
 110     if (text.isEmpty())
 111       continue;
 112     passFile += widget->objectName() + ": " + text + "\n";
 113   }
 114   passFile += ui->plainTextEdit->toPlainText();
 115   return passFile;
 116 }

 118 /**
 119  * @brief PasswordDialog::setTemplate set the template and create the fields.
 120  * @param rawFields
 121  */
 122 void PasswordDialog::setTemplate(QString rawFields) {
 123   fields = rawFields.split('\n');
 124   QWidget *previous = ui->checkBoxShow;
 125   foreach (QString field, fields) {
 126     if (field.isEmpty())
 127       continue;
 128     QLineEdit *line = new QLineEdit();
 129     line->setObjectName(field);
 130     ui->formLayout->addRow(new QLabel(field), line);
 131     setTabOrder(previous, line);
 132     previous = line;
 133   }
 134 }
