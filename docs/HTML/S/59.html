 141 #ifdef Q_OS_WIN

 148   DWORD size = sizeof(dst);
 149   DWORD r = WNetUseConnectionA(
 150       reinterpret_cast<HWND>(effectiveWinId()), &netres,
 151       webDavPassword.toUtf8().constData(), webDavUser.toUtf8().constData(),
 152       CONNECT_TEMPORARY | CONNECT_INTERACTIVE | CONNECT_REDIRECT, dst, &size,
 153       0);
 154   if (r == NO_ERROR) {
 155     passStore = dst;
 156   } else {
 157     char message[256] = {0};
 158     FormatMessageA(FORMAT_MESSAGE_FROM_SYSTEM, 0, r, 0, message,
 159                    sizeof(message), 0);
 160     ui->textBrowser->setTextColor(Qt::red);

 162                              " (0x" + QString::number(r, 16) + ")");
 163   }
 164 #else
 165   fusedav.start("fusedav -o nonempty -u \"" + webDavUser + "\" " + webDavUrl +
 166                 " \"" + passStore + '"');
 167   fusedav.waitForStarted();
 168   if (fusedav.state() == QProcess::Running) {
 169     QString pwd = webDavPassword;
 170     bool ok = true;
 171     if (pwd.isEmpty()) {
 172       pwd = QInputDialog::getText(this, tr("QtPass WebDAV password"),
 173                                   tr("Enter password to connect to WebDAV:"),
 174                                   QLineEdit::Password, "", &ok);
 175     }
 176     if (ok && !pwd.isEmpty()) {
 177       fusedav.write(pwd.toUtf8() + '\n');
 178       fusedav.closeWriteChannel();
 179       fusedav.waitForFinished(2000);
 180     } else {

 182     }
 183   }
 184   QString error = fusedav.readAllStandardError();
 185   int prompt = error.indexOf("Password:");
 186   if (prompt >= 0)
 187     error.remove(0, prompt + 10);
 188   if (fusedav.state() != QProcess::Running)
 189     error = tr("fusedav exited unexpectedly\n") + error;
 190   if (error.size() > 0) {
 191     ui->textBrowser->setTextColor(Qt::red);
 192     ui->textBrowser->setText(
 193         tr("Failed to start fusedav to connect WebDAV:\n") + error);
 194   }

 739 /**
 740  * @brief MainWindow::executeWrapper run an application, queue when needed.
 741  * @param app path to application to run
 742  * @param args required arguements
 743  * @param input optional input
 744  */
 745 void MainWindow::executeWrapper(QString app, QString args, QString input) {
 746   // qDebug() << app + " " + args;
 747   // Happens a lot if e.g. git binary is not set.
 748   // This will result in bogus "QProcess::FailedToStart" messages,
 749   // also hiding legitimate errors from the gpg commands.
 750   if (app.isEmpty()) {
 751     qDebug() << "Trying to execute nothing..";
 752     return;
 753   }
 754   // Convert to absolute path, just in case
 755   app = QDir(QCoreApplication::applicationDirPath()).absoluteFilePath(app);
 756   if (wrapperRunning) {
 757     execQueueItem item;
 758     item.app = app;
 759     item.args = args;
 760     item.input = input;
 761 
 762     execQueue->enqueue(item);
 763     qDebug() << item.app + "," + item.args + "," + item.input;
 764     return;
 765   }
 766   wrapperRunning = true;
 767   process->setWorkingDirectory(passStore);
 768   process->setEnvironment(env);
 769   clearTemplateWidgets();
 770   ui->textBrowser->clear();
 771   ui->textBrowser->setTextColor(Qt::black);
 772   enableUiElements(false);
 773   if (autoclearTimer != NULL) {
 774     autoclearTimer->stop();
 775     delete autoclearTimer;
 776     autoclearTimer = NULL;
 777   }
 778   process->start('"' + app + "\" " + args);
 779   if (!input.isEmpty())
 780     process->write(input.toUtf8());
 781   process->closeWriteChannel();
 782 }

 784 /**
 785  * @brief MainWindow::readyRead we have data
 786  */
 787 void MainWindow::readyRead(bool finished = false) {
 788   if (currentAction == PWGEN)
 789     return;
 790   QString output = "";
 791   QString error = "";
 792   if (currentAction != GPG_INTERNAL) {
 793     error = process->readAllStandardError();
 794     QByteArray processOutBytes = process->readAllStandardOutput();
 795     QTextCodec *codec = QTextCodec::codecForLocale();
 796     output = codec->toUnicode(processOutBytes);
 797     if (finished && currentAction == GPG) {
 798       lastDecrypt = output;
 799       QStringList tokens = output.split("\n");
 800 
 801       if (useClipboard != CLIPBOARD_NEVER && !output.isEmpty()) {
 802         setClippedPassword(tokens[0]);
 803         if (useClipboard == CLIPBOARD_ALWAYS)
 804           copyPasswordToClipboard();
 805         if (useAutoclearPanel) {
 806           QTimer::singleShot(1000 * autoclearPanelSeconds, this,
 807                              SLOT(clearPanel()));
 808         }
 809         if (hidePassword && !useTemplate) {
 810           tokens[0] = "***" + tr("Password hidden") + "***";
 811           output = tokens.join("\n");
 812         }
 813         if (hideContent)
 814           output = "***" + tr("Content hidden") + "***";
 815       }
 816 
 817       if (useTemplate && !hideContent) {
 818         while (ui->formLayout->count() > 0) {
 819           QLayoutItem *item = ui->formLayout->takeAt(0);
 820           delete item->widget();
 821           delete item;
 822         }
 823         QLineEdit *pass = new QLineEdit();
 824         pass->setText(tokens[0]);
 825         tokens.pop_front();
 826         if (hidePassword)
 827           pass->setEchoMode(QLineEdit::Password);
 828         pass->setReadOnly(true);
 829         ui->formLayout->addRow(pass);
 830 
 831         for (int j = 0; j < tokens.length(); ++j) {
 832           QString token = tokens.at(j);
 833           if (token.contains(':')) {
 834             int colon = token.indexOf(':');
 835             QString field = token.left(colon);
 836             if (templateAllFields || passTemplate.contains(field)) {
 837               QString value = token.right(token.length() - colon - 1);
 838               if (!passTemplate.contains(field) && value.startsWith("//"))
 839                 continue; // colon is probably from a url
 840               QLineEdit *line = new QLineEdit();
 841               line->setObjectName(field);
 842               line->setText(value);
 843               line->setReadOnly(true);
 844               ui->formLayout->addRow(new QLabel(field), line);
 845               tokens.removeAt(j);
 846               --j; // tokens.length() also got shortened by the remove..
 847             }
 848           }
 849         }
 850         if (ui->formLayout->count() == 0)
 851           ui->verticalLayoutPassword->setSpacing(0);
 852         else
 853           ui->verticalLayoutPassword->setSpacing(6);
 854         output = tokens.join("\n");
 855       } else {
 856         clearTemplateWidgets();
 857       }
 858       if (useAutoclearPanel) {
 859         autoclearPass = output;
 860         autoclearTimer = new QTimer(this);
 861         autoclearTimer->setSingleShot(true);
 862         autoclearTimer->setInterval(1000 * autoclearPanelSeconds);
 863         connect(autoclearTimer, SIGNAL(timeout()), this, SLOT(clearPanel()));
 864         autoclearTimer->start();
 865       }
 866     }
 867     output.replace(QRegExp("<"), "&lt;");
 868     output.replace(QRegExp(">"), "&gt;");
 869   } else {
 870     // qDebug() << process->readAllStandardOutput();
 871     // qDebug() << process->readAllStandardError();
 872     if (finished && 0 != keygen) {
 873       qDebug() << "Keygen Done";
 874       keygen->close();
 875       keygen = 0;
 876       // TODO(annejan) some sanity checking ?
 877     }
 878   }
 879 
 880   if (!error.isEmpty()) {
 881     if (currentAction == GIT) {
 882       // https://github.com/IJHack/qtpass/issues/111
 883       output = "<span style=\"color: darkgray;\">" + error + "</span><br />" +
 884                output;
 885     } else {
 886       output =
 887           "<span style=\"color: red;\">" + error + "</span><br />" + output;
 888     }
 889   }
 890 
 891   output.replace(QRegExp("((?:https?|ftp)://\\S+)"), "<a href=\"\\1\">\\1</a>");
 892   output.replace(QRegExp("\n"), "<br />");
 893   if (!ui->textBrowser->toPlainText().isEmpty())
 894     output = ui->textBrowser->toHtml() + output;
 895   ui->textBrowser->setHtml(output);
 896 }

1083 /**
1084  * @brief MainWindow::getRecipientList return list op gpg-id's to encrypt for
1085  * @param for_file which file (folder) would you like recepients for
1086  * @return recepients gpg-id contents
1087  */
1088 QStringList MainWindow::getRecipientList(QString for_file) {
1089   QDir gpgIdPath(QFileInfo(for_file.startsWith(passStore)
1090                                ? for_file
1091                                : passStore + for_file)
1092                      .absoluteDir());
1093   bool found = false;
1094   while (gpgIdPath.exists() && gpgIdPath.absolutePath().startsWith(passStore)) {
1095     if (QFile(gpgIdPath.absoluteFilePath(".gpg-id")).exists()) {
1096       found = true;
1097       break;
1098     }
1099     if (!gpgIdPath.cdUp())
1100       break;
1101   }
1102   QFile gpgId(found ? gpgIdPath.absoluteFilePath(".gpg-id")
1103                     : passStore + ".gpg-id");
1104   if (!gpgId.open(QIODevice::ReadOnly | QIODevice::Text))
1105     return QStringList();
1106   QStringList recipients;
1107   while (!gpgId.atEnd()) {
1108     QString recipient(gpgId.readLine());
1109     recipient = recipient.trimmed();
1110     if (!recipient.isEmpty())
1111       recipients += recipient;
1112   }
1113   return recipients;
1114 }

1356 /**
1357  * @brief MainWindow::listKeys list users
1358  * @param keystring
1359  * @param secret list private keys
1360  * @return QList<UserInfo> users
1361  */
1362 QList<UserInfo> MainWindow::listKeys(QString keystring, bool secret) {
1363   waitFor(5);
1364   QList<UserInfo> users;
1365   currentAction = GPG_INTERNAL;
1366   QString listopt = secret ? "--list-secret-keys " : "--list-keys ";
1367   executeWrapper(gpgExecutable,
1368                  "--no-tty --with-colons " + listopt + keystring);
1369   process->waitForFinished(2000);
1370   if (process->exitStatus() != QProcess::NormalExit)
1371     return users;
1372   QByteArray processOutBytes = process->readAllStandardOutput();
1373   QTextCodec *codec = QTextCodec::codecForLocale();
1374   QString processOutString = codec->toUnicode(processOutBytes);
1375   QStringList keys = QString(processOutString)
1376                          .split(QRegExp("[\r\n]"), QString::SkipEmptyParts);
1377   UserInfo current_user;
1378   foreach (QString key, keys) {
1379     QStringList props = key.split(':');
1380     if (props.size() < 10)
1381       continue;
1382     if (props[0] == (secret ? "sec" : "pub")) {
1383       if (!current_user.key_id.isEmpty())
1384         users.append(current_user);
1385       current_user = UserInfo();
1386       current_user.key_id = props[4];
1387       current_user.name = props[9].toUtf8();
1388       current_user.validity = props[1][0].toLatin1();
1389       current_user.created.setTime_t(props[5].toInt());
1390       current_user.expiry.setTime_t(props[6].toInt());
1391     } else if (current_user.name.isEmpty() && props[0] == "uid") {
1392       current_user.name = props[9];
1393     }
1394   }
1395   if (!current_user.key_id.isEmpty())
1396     users.append(current_user);
1397   return users;
1398 }

1410 /**
1411  * @brief MainWindow::on_usersButton_clicked edit users for the current folder,
1412  * gets lists and opens UserDialog.
1413  */
1414 void MainWindow::on_usersButton_clicked() {
1415   QList<UserInfo> users = listKeys();
1416   if (users.size() == 0) {
1417     QMessageBox::critical(this, tr("Can not get key list"),
1418                           tr("Unable to get list of available gpg keys"));
1419     return;
1420   }
1421   QList<UserInfo> secret_keys = listKeys("", true);
1422   foreach (const UserInfo &sec, secret_keys) {
1423     for (QList<UserInfo>::iterator it = users.begin(); it != users.end(); ++it)
1424       if (sec.key_id == it->key_id)
1425         it->have_secret = true;
1426   }
1427   QList<UserInfo> selected_users;
1428   QString dir = currentDir.isEmpty()
1429                     ? getDir(ui->treeView->currentIndex(), false)
1430                     : currentDir;
1431   int count = 0;
1432   QString recipients =
1433       getRecipientString(dir.isEmpty() ? "" : dir, " ", &count);
1434   if (!recipients.isEmpty())
1435     selected_users = listKeys(recipients);
1436   foreach (const UserInfo &sel, selected_users) {
1437     for (QList<UserInfo>::iterator it = users.begin(); it != users.end(); ++it)
1438       if (sel.key_id == it->key_id)
1439         it->enabled = true;
1440   }
1441   if (count > selected_users.size()) {
1442     // Some keys seem missing from keyring, add them separately
1443     QStringList recipients = getRecipientList(dir.isEmpty() ? "" : dir);
1444     foreach (const QString recipient, recipients) {
1445       if (listKeys(recipient).size() < 1) {
1446         UserInfo i;
1447         i.enabled = true;
1448         i.key_id = recipient;
1449         i.name = " ?? " + tr("Key not found in keyring");
1450         users.append(i);
1451       }
1452     }
1453   }
1454   UsersDialog d(this);
1455   d.setUsers(&users);
1456   if (!d.exec()) {
1457     d.setUsers(NULL);
1458     return;
1459   }
1460   d.setUsers(NULL);
1461 
1462   if (usePass) {
1463     QString gpgIds = "";
1464     foreach (const UserInfo &user, users) {
1465       if (user.enabled) {
1466         gpgIds += user.key_id + " ";
1467       }
1468     }
1469     executePass("init --path=" + dir + " " + gpgIds);
1470   } else {
1471     QString gpgIdFile = dir + ".gpg-id";
1472     QFile gpgId(gpgIdFile);
1473     bool addFile = false;
1474     if (addGPGId) {
1475       QFileInfo checkFile(gpgIdFile);
1476       if (!checkFile.exists() || !checkFile.isFile())
1477         addFile = true;
1478     }
1479     if (!gpgId.open(QIODevice::WriteOnly | QIODevice::Text)) {
1480       QMessageBox::critical(this, tr("Cannot update"),
1481                             tr("Failed to open .gpg-id for writing."));
1482       return;
1483     }
1484     bool secret_selected = false;
1485     foreach (const UserInfo &user, users) {
1486       if (user.enabled) {
1487         gpgId.write((user.key_id + "\n").toUtf8());
1488         secret_selected |= user.have_secret;
1489       }
1490     }
1491     gpgId.close();
1492     if (!secret_selected) {
1493       QMessageBox::critical(
1494           this, tr("Check selected users!"),
1495           tr("None of the selected keys have a secret key available.\n"
1496              "You will not be able to decrypt any newly added passwords!"));
1497     }
1498 
1499     if (!useWebDav && useGit && !gitExecutable.isEmpty()) {
1500       if (addFile)
1501         executeWrapper(gitExecutable, "add \"" + gpgIdFile + '"');
1502       QString path = gpgIdFile;
1503       path.replace(QRegExp("\\.gpg$"), "");
1504       executeWrapper(gitExecutable, "commit \"" + gpgIdFile + "\" -m \"Added " +
1505                                         path + " using QtPass.\"");
1506       if (autoPush)
1507         on_pushButton_clicked();
1508     }
1509 
1510     reencryptPath(dir);
1511   }
1512 }

1875  * generator
1876  * @return the password
1877  */

1879 QString MainWindow::generatePassword(int length, clipBoardType selection) {
1880   QString passwd;
1881   if (usePwgen) {
1882     waitFor(2);
1883     // --secure goes first as it overrides --no-* otherwise
1884     QString args = QString("-1 ") + (lessRandom ? "" : "--secure ") +
1885                    (avoidCapitals ? "--no-capitalize " : "--capitalize ") +
1886                    (avoidNumbers ? "--no-numerals " : "--numerals ") +
1887                    (useSymbols ? "--symbols " : "") + QString::number(length);
1888     currentAction = PWGEN;
1889     executeWrapper(pwgenExecutable, args);
1890     process->waitForFinished(1000);
1891     if (process->exitStatus() == QProcess::NormalExit)
1892       passwd =
1893           QString(process->readAllStandardOutput()).remove(QRegExp("[\\n\\r]"));
1894     else
1895       qDebug() << "pwgen fail";
1896   } else {
1897     int charsetLength = pwdConfig.Characters[selection].length();
1898     if (charsetLength > 0) {
1899       for (int i = 0; i < length; ++i) {
1900         int index = qrand() % charsetLength;
1901         QChar nextChar = pwdConfig.Characters[selection].at(index);
1902         passwd.append(nextChar);
1903       }
1904     } else {
1905       QMessageBox::critical(
1906           this, tr("No characters chosen"),
1907           tr("Can't generate password, there are no characters to choose from "
1908              "set in the configuration!"));
1909     }
1910   }
1911   return passwd;
1912 }

1914 /**
1915  * @brief MainWindow::waitFor wait until process->atEnd and execQueue->isEmpty
1916  * or timeout after x-seconds
1917  * @param seconds
1918  */
1919 void MainWindow::waitFor(int seconds) {
1920   QDateTime current = QDateTime::currentDateTime();
1921   uint stop = current.toTime_t() + seconds;
1922   while (!process->atEnd() || !execQueue->isEmpty()) {
1923     current = QDateTime::currentDateTime();
1924     if (stop < current.toTime_t()) {
1925       QMessageBox::critical(
1926           this, tr("Timed out"),
1927           tr("Can't start process, previous one is still running!"));
1928       return;
1929     }
1930     Util::qSleep(100);
1931   }
1932 }
