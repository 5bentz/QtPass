175 void Pass::executeWrapper(QString app, QString args, QString input) {
176  // qDebug() << app + " " + args;
177  // Happens a lot if e.g. git binary is not set.
178  // This will result in bogus "QProcess::FailedToStart" messages,
179  // also hiding legitimate errors from the gpg commands.
180  if (app.isEmpty()) {
181  qDebug() << "Trying to execute nothing..";
182  return;
183  }
184  // Convert to absolute path, just in case
185  app = QDir(QCoreApplication::applicationDirPath()).absoluteFilePath(app);
186  if (wrapperRunning) {
187  execQueueItem item;
188  item.app = app;
189  item.args = args;
190  item.input = input;
191 
192  execQueue.enqueue(item);
193  qDebug() << item.app + "," + item.args + "," + item.input;
194  return;
195  }
196  wrapperRunning = true;
197  process.setWorkingDirectory(QtPassSettings::getPassStore());
198  process.setEnvironment(env);
199  emit startingExecuteWrapper();
200  process.start('"' + app + "\" " + args);
201  if (!input.isEmpty())
202  process.write(input.toUtf8());
203  process.closeWriteChannel();
204 }
